  <script>
    // Data lists (teams and player examples)
    const teams = [
      'Chennai Super Kings','Delhi Capitals','Gujarat Titans','Kolkata Knight Riders',
      'Lucknow Super Giants','Mumbai Indians','Punjab Kings','Rajasthan Royals',
      'Royal Challengers Bengaluru','Sunrisers Hyderabad'
    ];

    const players = [
      'MS Dhoni','Virat Kohli','Rohit Sharma','Jasprit Bumrah','KL Rahul','Shubman Gill',
      'Jos Buttler','Rishabh Pant','Pat Cummins','Mitchell Starc','Ben Stokes','Sam Curran',
      'Ishan Kishan','Ruturaj Gaikwad','Kane Williamson','David Warner','Bhuvneshwar Kumar',
      'Mohammed Shami','Yuzvendra Chahal','Harshal Patel'
    ];

    // Helper to populate selects
    function populateSelect(id, items){
      const sel = document.getElementById(id);
      sel.innerHTML = items.map(it => `<option value="${it}">${it}</option>`).join('\n');
    }

    // Initial population for all forms
    function initAll() {
      // winner
      populateSelect('winnerTeam1', teams);
      populateSelect('winnerTeam2', teams);
      // runner
      populateSelect('runnerTeam1', teams);
      populateSelect('runnerTeam2', teams);
      // purple/orange/expensive players
      populateSelect('purplePlayer1', players);
      populateSelect('purplePlayer2', players);
      populateSelect('orangePlayer1', players);
      populateSelect('orangePlayer2', players);
      populateSelect('expPlayer1', players);
      populateSelect('expPlayer2', players);

      // wire up dynamic predicted selects
      ['winner','runner'].forEach(prefix=>{
        const t1=document.getElementById(prefix+'Team1');
        const t2=document.getElementById(prefix+'Team2');
        const pred=document.getElementById(prefix+'Predicted');
        function update(){
          pred.innerHTML = `<option value="">Select your predicted team</option>`+
            `<option value="${t1.value}">${t1.value}</option>`+
            `<option value="${t2.value}">${t2.value}</option>`;
        }
        t1.addEventListener('change',update);
        t2.addEventListener('change',update);
        update();
      });

      // players predicted selects
      function bindPlayers(p1id,p2id,predid){
        const p1=document.getElementById(p1id);
        const p2=document.getElementById(p2id);
        const predicted=document.getElementById(predid);
        function update(){
          predicted.innerHTML = `<option value="">Select one player</option>`+
            `<option value="${p1.value}">${p1.value}</option>`+
            `<option value="${p2.value}">${p2.value}</option>`;
        }
        p1.addEventListener('change',update);
        p2.addEventListener('change',update);
        update();
      }
      bindPlayers('purplePlayer1','purplePlayer2','purplePredicted');
      bindPlayers('orangePlayer1','orangePlayer2','orangePredicted');
      bindPlayers('expPlayer1','expPlayer2','expPredicted');
    }

    // Simple show/hide page sections
    function show(id){
      document.querySelectorAll('section').forEach(s=>s.classList.add('hidden'));
      document.getElementById(id).classList.remove('hidden');
      window.scrollTo({top:0,behavior:'smooth'});
    }

    // Landing open button -> registration
    document.getElementById('openSite').addEventListener('click',()=>show('registration'));
    document.getElementById('backToLanding').addEventListener('click',(e)=>{e.preventDefault();show('landing')});
    document.getElementById('continueBtn').addEventListener('click',()=>{
      const name = document.getElementById('regName').value.trim();
      if(!name){alert('Please enter your name.');return}
      // store or greet (you can extend)
      document.getElementById('registration').querySelector('h2').innerText = `Registration Page — Hello, ${name}`;
      show('options');
    });
    document.getElementById('backToRegistration').addEventListener('click',(e)=>{e.preventDefault();show('registration')});

    // Options buttons -> respective forms
    document.querySelectorAll('#options .btn').forEach(btn=>{
      btn.addEventListener('click',()=>{
        const target = btn.dataset.target;
        show(target);
      });
    });

    // Back buttons inside prediction cards
    document.getElementById('backFromWinner').addEventListener('click',()=>show('options'));
    document.getElementById('backFromRunner').addEventListener('click',()=>show('options'));
    document.getElementById('backFromPurple').addEventListener('click',()=>show('options'));
    document.getElementById('backFromOrange').addEventListener('click',()=>show('options'));
    document.getElementById('backFromExp').addEventListener('click',()=>show('options'));

    // Submit handlers: all show the same result message as in your code
    function submitHandler(predId, resultId){
      const sel = document.getElementById(predId);
      const res = document.getElementById(resultId);
      if(!sel.value){res.innerText='Please select a prediction.';return}
      res.innerText = `You chose ${sel.value} on a 60:40 ratio and the result will come after 10:00 PM. Stay tuned to know if you won or not.`;
    }

    document.getElementById('submitWinner').addEventListener('click',()=>submitHandler('winnerPredicted','winnerResult'));
    document.getElementById('submitRunner').addEventListener('click',()=>submitHandler('runnerPredicted','runnerResult'));
    document.getElementById('submitPurple').addEventListener('click',()=>submitHandler('purplePredicted','purpleResult'));
    document.getElementById('submitOrange').addEventListener('click',()=>submitHandler('orangePredicted','orangeResult'));
    document.getElementById('submitExp').addEventListener('click',()=>submitHandler('expPredicted','expResult'));

    // initialize
    initAll();
    // show landing at start
    show('landing');
